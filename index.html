const tabs = (headerSelector,tabSelector,tabContentSelector,classActive,display = 'block') => {
    const header = document.querySelector(headerSelector),
            content = document.querySelectorAll(tabContentSelector),
            tab = document.querySelectorAll(tabSelector);

    function hide() {
        content.forEach((item)=>{
            item.style.display = 'none';
        });

        tab.forEach((item)=>{
            item.classList.remove(classActive);
        });
    }
    function show(i = 0) {
        content[i].style.display = display;
        tab[i].classList.add(classActive);
    }

    header.addEventListener('click',(e)=>{
        const target = e.target;
        if( target &&
            (target.classList.contains(tabSelector.replace(/\./, '')) ||
            target.parentNode.classList.contains(tabSelector.replace(/\./, '')))) {
            tab.forEach((item,i)=>{
                if(item === target || item == target.parentNode){
                    hide();
                    show(i); 
                }
            });
        }
    });
    hide();
    show();
};

export default tabs;
